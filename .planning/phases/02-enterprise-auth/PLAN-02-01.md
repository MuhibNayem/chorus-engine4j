# Plan 02-01: Schema & Domain Models

## Goal
Create the database schema (Flyway V7) and Java domain models for all Phase 2 entities: per-tenant OAuth2/SAML IdP configurations, SCIM provisioning tokens, and the updated `users` table with auth source tracking. This plan is pure infrastructure — no controllers, no filters, no security wiring.

## must_haves
- Flyway V7 migration creates `tenant_oauth_configs`, `tenant_saml_configs`, `scim_tokens` tables and updates `users` with `auth_source` and a `LOWER(email)` unique constraint.
- All new domain models are Java records with `@NonNull`/`@Nullable` annotations following existing patterns.
- All new repositories use pure JDBC with JdbcTemplate, ON CONFLICT upserts, and hand-written RowMappers.
- Spring Security OAuth2, SAML2, and JOSE dependencies are added to build.gradle.kts.
- `./gradlew :chorus-observe-server:compileJava` passes.

CREATE TABLE tenant_oauth_configs (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    scopes          JSONB NOT NULL DEFAULT '[]'::jsonb,

    enabled         BOOLEAN NOT NULL DEFAULT true,

    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE (tenant_id, provider_name)

-- SAML IdP configurations per tenant (multiple IdPs supported)
CREATE TABLE tenant_saml_configs (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       VARCHAR(64) NOT NULL REFERENCES tenants(tenant_id) ON DELETE CASCADE,
    provider_name   VARCHAR(64) NOT NULL,
    entity_id       VARCHAR(512) NOT NULL,
    sign_on_url     VARCHAR(512) NOT NULL,
    signing_cert_thumbprint VARCHAR(128) NOT NULL,
    metadata_url    VARCHAR(512),
    acs_url         VARCHAR(512) NOT NULL,
    default_role    VARCHAR(64) NOT NULL DEFAULT 'VIEWER',
    enabled         BOOLEAN NOT NULL DEFAULT true,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE (tenant_id, provider_name)
);

CREATE INDEX idx_tenant_saml_configs_tenant ON tenant_saml_configs(tenant_id);

-- SCIM provisioning tokens (separate from api_keys)
CREATE TABLE scim_tokens (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       VARCHAR(64) NOT NULL REFERENCES tenants(tenant_id) ON DELETE CASCADE,
    name            VARCHAR(128) NOT NULL,
    token_hash      VARCHAR(256) NOT NULL UNIQUE,
    scopes          JSONB NOT NULL DEFAULT '[]'::jsonb,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    expires_at      TIMESTAMPTZ,
    revoked_at      TIMESTAMPTZ
);

CREATE INDEX idx_scim_tokens_tenant ON scim_tokens(tenant_id);
CREATE INDEX idx_scim_tokens_hash ON scim_tokens(token_hash);

-- Add auth_source to users for tracking login method
ALTER TABLE users ADD COLUMN IF NOT EXISTS auth_source VARCHAR(16) NOT NULL DEFAULT 'LOCAL';

-- LOWER(email) unique constraint per tenant for duplicate prevention (SCIM-07)
CREATE UNIQUE INDEX idx_users_tenant_lower_email ON users(tenant_id, LOWER(email));

Create `chorus-observe-server/src/main/java/com/chorus/observe/model/ScimToken.java`:

```java
package com.chorus.observe.model;

import org.jspecify.annotations.NonNull;
import org.jspecify.annotations.Nullable;

import java.time.Instant;
import java.util.List;
import java.util.Objects;
import java.util.UUID;

public record ScimToken(
    @Nullable UUID id,
    @NonNull String tenantId,
    @NonNull String name,
    @NonNull String tokenHash,
    @NonNull List<String> scopes,
    @NonNull Instant createdAt,
    @Nullable Instant expiresAt,
    @Nullable Instant revokedAt
) {
    public ScimToken {
        Objects.requireNonNull(tenantId);
        Objects.requireNonNull(name);
        Objects.requireNonNull(tokenHash);
        Objects.requireNonNull(scopes);
        Objects.requireNonNull(createdAt);
    }

    public boolean isExpired() {
        return expiresAt != null && Instant.now().isAfter(expiresAt);
    }

    public boolean isRevoked() {
        return revokedAt != null;
    }

    public boolean isActive() {
        return !isExpired() && !isRevoked();
    }
}
```
</action>
